# Destructuring Assignment

## src/destructuringassignment.ts

```ts
var { mycolor, ...remaining } = {
    "mycolor": "blue",
    "position": "Next",
    "myname": "John"
}

console.log( mycolor );

let [x,,y] = [1, 2, 3,4];
console.log("x:"+x + " y:" + y);

let [x1,y1,...remaining1] = [1, 2, 3,4];
console.log("x:"+x + " y:" + y, remaining1);

```
